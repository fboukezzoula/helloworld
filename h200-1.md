Voici un Terraform complet pour ton POC avec les contraintes que tu as données :

✅ VNet existant en francecentral

✅ création d’un VNet GPU en southafricanorth

✅ VNet Peering entre les deux

✅ VM H200 (Standard_ND96isr_H200_v5)

✅ aucune IP publique

✅ Managed Identity

✅ installation automatique des drivers NVIDIA + CUDA

✅ Accelerated Networking

Je suppose que ton VNet existant est dans le même resource group.

# Structure

```
FranceCentral
   existing-vnet
       subnet rfc-vm-h200

SouthAfricaNorth
   vnet-h200
       subnet h200-subnet
           VM ND96isr_H200_v5

VNet Peering
```

# Terraform

```hcl
terraform {
  required_version = ">=1.5"

  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~>3.100"
    }
  }
}

provider "azurerm" {
  features {}
}

#####################################
# VARIABLES
#####################################

variable "resource_group_name" {}
variable "existing_vnet_name" {}
variable "ssh_public_key" {}

variable "location_gpu" {
  default = "southafricanorth"
}

#####################################
# RESOURCE GROUP
#####################################

data "azurerm_resource_group" "rg" {
  name = var.resource_group_name
}

#####################################
# EXISTING VNET (France Central)
#####################################

data "azurerm_virtual_network" "fr_vnet" {
  name                = var.existing_vnet_name
  resource_group_name = data.azurerm_resource_group.rg.name
}

#####################################
# GPU VNET (South Africa North)
#####################################

resource "azurerm_virtual_network" "gpu_vnet" {

  name                = "vnet-h200-sa"
  location            = var.location_gpu
  resource_group_name = data.azurerm_resource_group.rg.name

  address_space = ["10.200.0.0/16"]
}

resource "azurerm_subnet" "gpu_subnet" {

  name                 = "h200-subnet"
  resource_group_name  = data.azurerm_resource_group.rg.name
  virtual_network_name = azurerm_virtual_network.gpu_vnet.name

  address_prefixes = ["10.200.1.0/24"]
}

#####################################
# VNET PEERING
#####################################

resource "azurerm_virtual_network_peering" "fr_to_gpu" {

  name                      = "fr-to-h200"
  resource_group_name       = data.azurerm_resource_group.rg.name
  virtual_network_name      = data.azurerm_virtual_network.fr_vnet.name
  remote_virtual_network_id = azurerm_virtual_network.gpu_vnet.id

  allow_virtual_network_access = true
}

resource "azurerm_virtual_network_peering" "gpu_to_fr" {

  name                      = "h200-to-fr"
  resource_group_name       = data.azurerm_resource_group.rg.name
  virtual_network_name      = azurerm_virtual_network.gpu_vnet.name
  remote_virtual_network_id = data.azurerm_virtual_network.fr_vnet.id

  allow_virtual_network_access = true
}

#####################################
# NETWORK INTERFACE (NO PUBLIC IP)
#####################################

resource "azurerm_network_interface" "gpu_nic" {

  name                = "nic-h200"
  location            = var.location_gpu
  resource_group_name = data.azurerm_resource_group.rg.name

  accelerated_networking_enabled = true

  ip_configuration {
    name                          = "internal"
    subnet_id                     = azurerm_subnet.gpu_subnet.id
    private_ip_address_allocation = "Dynamic"
  }
}

#####################################
# H200 VM
#####################################

resource "azurerm_linux_virtual_machine" "h200_vm" {

  name                = "vm-h200-poc"
  resource_group_name = data.azurerm_resource_group.rg.name
  location            = var.location_gpu

  size = "Standard_ND96isr_H200_v5"

  admin_username = "azureuser"

  network_interface_ids = [
    azurerm_network_interface.gpu_nic.id
  ]

  disable_password_authentication = true

  admin_ssh_key {
    username   = "azureuser"
    public_key = var.ssh_public_key
  }

  #################################
  # Managed Identity
  #################################

  identity {
    type = "SystemAssigned"
  }

  #################################
  # OS Disk
  #################################

  os_disk {
    caching              = "ReadWrite"
    storage_account_type = "Premium_LRS"
  }

  #################################
  # Ubuntu Image
  #################################

  source_image_reference {
    publisher = "Canonical"
    offer     = "0001-com-ubuntu-server-jammy"
    sku       = "22_04-lts"
    version   = "latest"
  }

  boot_diagnostics {}
}

#####################################
# NVIDIA GPU DRIVER + CUDA
#####################################

resource "azurerm_virtual_machine_extension" "nvidia_driver" {

  name                 = "nvidiaGpuDriver"
  virtual_machine_id   = azurerm_linux_virtual_machine.h200_vm.id
  publisher            = "Microsoft.HpcCompute"
  type                 = "NvidiaGpuDriverLinux"
  type_handler_version = "1.9"

  auto_upgrade_minor_version = true

  settings = <<SETTINGS
{
  "updateOS": true
}
SETTINGS
}
```
